CREATE DATABASE SpaceBook;

USE  SpaceBook;

CREATE TABLE 
	viagem (
FOREIGN KEY (id_passageiro) REFERENCES passageiro (id_passageiro),
FOREIGN KEY (id_piloto) REFERENCES piloto (id_piloto),
id_viagem INT AUTO_INCREMENT PRIMARY KEY
data_viagem VARCHAR(10) NOT NULL,
destino_viagem VARCHAR(10) NOT NULL,
preco_viagem INT NOT NULL,

);

CREATE TABLE
	passageiro (
id_passageiro INT AUTO_INCREMENT PRIMARY KEY
email_passageiro VARCHAR(30) NOT NULL,
senha_passageiro VARCHAR(20) NOT NULL,
nome_passageiro VARCHAR(30) NOT NULL,
endereco_passageiro VARCHAR(30) NOT NULL,
cpf_passageiro VARCHAR(11) NOT NULL,
telefone_passageiro VARCHAR(15) NOT NULL,
);

CREATE TABLE
	piloto (
FOREIGN KEY (id_viagem) REFERENCES viagem (id_viagem),
id_piloto INT AUTO_INCREMENT PRIMARY KEY
certificacao_piloto VARCHAR(50) NOT NULL,
especialidade_piloto VARCHAR(30) NOT NULL,
nome_piloto VARCHAR(30) NOT NULL,
);
